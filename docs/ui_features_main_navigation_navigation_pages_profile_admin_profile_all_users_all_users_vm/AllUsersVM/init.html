Future<void> init() async {
  isLoading = true;
  notifyListeners();
  try {
    final currentUser = await loginRepository.getUserCurrentlyLogged();
    amcaUsers = await usersRepository.getAllUsers()
      ..removeWhere(
          (element) => element.identification == currentUser.identification);
  } catch (e) {
  } finally {
    isLoading = false;
    notifyListeners();
  }
}
