@override
Future<void> recoverPassword(String email) async {
  try {
    return await _firebaseAuth
        .sendPasswordResetEmail(email: email);
  } on FirebaseAuthException catch (e) {
    throw AppException(
      message: e.message,
      codeError: e.code,
    );
  } catch (e) {
    throw AppException(
      codeError: Constants.generalError,
    );
  }
}
