void decreaseDate() {
    if (dateSelectedType == DateSelectedType.month) {
      currentDateTime = currentDateTime.subtract(months: 1);
    }
    if (dateSelectedType == DateSelectedType.semester) {
      currentDateTime = currentDateTime.subtract(months: 6);
    }

    notifyListeners();
  }
